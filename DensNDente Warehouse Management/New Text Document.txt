public bool DeleteByInvoiceDetailId(int id)
        {

            try
            {
                var result = (from r in repository.tblSaleInvoiceDetails
                              where r.InvoiceId == id
                              select r);
                

                 foreach (var detail in result)
                {
                    detail.Deleted = true;
                   
                }
                repository.SaveChanges();
                return true;
            }
            catch (Exception)
            {
                return false;
            }
        }
